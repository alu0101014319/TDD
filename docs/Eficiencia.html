    # File lib/feeding/plato.rb, line 145
def get_gasesA
        aux_gases = @alimentos.get_head()
        aux_gr = @gramos.get_head()
        gases_t = 0
        while !aux_gases.nil?
                gases_t += (aux_gases[:value].porcion_gases(aux_gr[:value]))
                aux_gases = aux_gases[:next]
                aux_gr = aux_gr[:next]
        end
        gases_t.round(2)
end

def hn
        if (get_kcal < 670)
                v_kcal = 1
        elsif (get_kcal > 830)
                v_kcal = 3
        else
                v_kcal = 2
        end

        if (get_gasesA < 0.8)
                v_gases = 1
        elsif (get_gasesA > 1.2)
                v_gases = 3
        else
                v_gases = 1
        end
        
        ((v_kcal + v_gases)/2.0).round(2)
end
